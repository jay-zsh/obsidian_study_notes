
# 最小公倍数与最大公约数

```c++
int my_gcd(int m,int n)
{
	if(m>n)swap(m,n);
	int temp;
	while(n!=0){
		temp = m%n;
		m = n;
		n = temp;
	}
	return m;
}
```


```c++
int my_lcm(int m,int n){
	return abs(m/my_gcd(m,n)*n);
}
```

```c++
//prime1.0

#include <iostream>
#define endl '\n'

typedef long long ll;
using namespace std;

bool is_prime(int n)
{
	if (n<=1)return false;
	if(n<=3)return true;
	if(n%2==0||n%3==0)return false;
	
	for(int i=5;i*i<n;i+=5)
	{
		if(n%i==0||(n+1)%i==0)return false;
	 } 
	return true;
} 

int main()
{
	ios::sync_with_stdio(0);
	cin.tie(0);
	cout.tie(0);
	
	if(is_prime(3))cout << "Yes" <<endl;

	
 }
```
